\section{Integration of Functions}

\subsection{Basics}

\subsubsection{Definition of Riemann Sums and Integrals}

We have learnt before that the integral
\[
    \int_{a}^{b} f\para{x} \Diff x
\]
gives the signed area under the graph of \(f\), for \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\). We now aim to quantify it rigorously.

We may attempt to approximate the function using rectangles -- specifically using rectangles \emph{taller} than the function to \emph{overestimate}, and \emph{lower} than the function to \emph{underestimate}. Ideally, if we make the width of the split smaller and smaller, however we split it, that the overestimate and the underestimate will eventually `converge' to the same limit -- which makes the function \emph{integrable}. (Warning: it is hard to quantify `smaller and smaller'!)

\begin{definition}[Upper and Lower Riemann Sums]
    \label{def:upper-lower-riemann-sum}%
    Let \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) be bounded. Let a \emph{partition/dissection} \(\partition = \brce{x_0, x_1, \ldots, x_n}\) with \(a = x_0 < x_1 < \cdots x_n = b\) be given. Then we let
    \[
        \upperSum \para{f, \partition} = \sum_{j = 1}^{n} \para{\sup_{x \in \ccintv{x_{j - 1}}{x_{j}}} f\para{x}} \cdot \para{x_j - x_{j - 1}}
    \]
    as the \emph{Upper Riemann Sum}, and similarly,
    \[
        \lowerSum \para{f, \partition} = \sum_{j = 1}^{n} \para{\inf_{x \in \ccintv{x_{j - 1}}{x_{j}}} f\para{x}} \cdot \para{x_j - x_{j - 1}}
    \]
    as the \emph{Lower Riemann Sum}, of the function \(f\) associated to the partition \(\partition\).
\end{definition}

\begin{example}[Riemann Sums]
    \label{ex:riemann-sums}%
    Consider the function
    \[
        f\para{x} = \begin{cases}
            1, & x \in \QQ, \\
            0, & x \in \RR \setminus \QQ
        \end{cases}
    \]
    on \(\ccintv{0}{1}\). For any \(\partition\) we take, we will have both rationals and irrationals in \(I_j = \ccintv{x_{j - 1}}{x_j}\), and hence we always have
    \[
        \upperSum \para{f, \partition} = 1, \quad \lowerSum \para{f, \partition} = 0.
    \]
\end{example}

We assumed that \(f\) is bounded, and say \(\sup_{x \in \ccintv{a}{b}} \abs{f} = k\), then
\begin{align*}
    \upperSum\para{f, \partition} &= \sum_{j = 1}^{n} \para{\sup_{\ccintv{a}{b}} f} \para{x_j - x_{j - 1}} \\
    &\leq k \sum_{j = 1}^{n} \para{x_j - x_{j - 1}} \\
    &= k \para{b - a},
\end{align*}
and similarly,
\[
    \lowerSum\para{f, \partition} \geq -k \para{b - a}.
\]

Also, we have \(\upperSum\para{f, \partition} \geq \lowerSum\para{f, \partition}\). Therefore, the sets
\[
    \set{\upperSum\para{f, \partition}}{\partition \text{ is a partition of }\ccintv{a}{b}}
\]
and
\[
    \set{\lowerSum\para{f, \partition}}{\partition \text{ is a partition of }\ccintv{a}{b}}
\]
are both bounded, non-empty sets.

\begin{definition}[Upper and Lower Integrals]
    \label{def:upper-lower-int}%
    Let \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) be bounded. We say
    \[
        \upperI \para{f} = \upperInt_{a}^{b} f \para{x} \Diff x = \inf_{\partition\text{ is a partition of }\ccintv{a}{b}} \upperSum \para{f, \partition}
    \]
    be the \emph{upper integral} of \(f\), and similarly,
    \[
        \lowerI \para{f} = \lowerInt_{a}^{b} f \para{x} \Diff x \sup_{\partition\text{ is a partition of }\ccintv{a}{b}} \lowerSum \para{f, \partition}
    \]
    be the \emph{lower integral} of \(f\).
\end{definition}

Note we actually didn't take a `limit' as the width tends to zero -- that is slightly tougher to formalise (it is possible to do so, as the limit of a mesh), but this works just as well (since narrowing the widths seems to decrease the error, which makes it smaller for overestimates, and larger for underestimates) -- but note because of this, lots of properties of the integral is \textbf{not} a direct consequence of the properties of the limit, which is a common misconception.

\begin{definition}[Riemann Integral]
    \label{def:riemann-integral}%
    We say \(f\) is \emph{Riemann Integrable} if \(\upperI \para{f} = \lowerI \para{f}\), and we denote the \emph{Riemann Integral} as
    \[
        I\para{f} = \int_{a}^{b} f\para{x} \Diff x = \upperI\para{f} = \lowerI\para{f}.
    \]
\end{definition}

\begin{remark}
    We define by convention
    \[
        \int_{a}^{b} f = - \int_{b}^{a} f.
    \]
\end{remark}

\begin{example}[Non-Riemann Integrable Function]
    \label{ex:non-riemann-integrable-fn}%
    Continuing from \autoref{ex:riemann-sums}, we note that \(\upperI\para{f} = 1\) while \(\lowerI\para{f} = 0\). Hence, this function is not Riemann integrable.
    
    Just as a remark, it is Lebesgue integrable using the Lebesgue measure, and the result is \(0\) -- intuitively, there are `way too few' rational numbers (and this is saying \(\QQ\) has zero Lebesgue measure, since it is countable).
\end{example}

The coarsest possible partition for \(\ccintv{a}{b}\) is the partition \(\brce{a, b}\). Our intuition is that a finer partition leads to a better estimate for the area.

\begin{lemma}[Refinements provide Better Estimates]
    \label{lem:refinement-better-estimate}%
    Let \(\partition, \partition'\) be two partitions of \(\ccintv{a}{b}\), with \(\partition' \supseteq \partition\) (i.e., \(\partition'\) is a \emph{refinement} of \(\partition\)). Then we have
    \[
        \lowerSum\para{f, \partition} \leq \lowerSum\para{f, \partition'} \leq \upperSum\para{f, \partition'} \leq \upperSum\para{f, \partition}.
    \]
\end{lemma}

\begin{proof}
    We show this by induction on refining the partition point-by-point.
    
    Take \(\partition = \brce{x_0, x_1, \ldots, x_n}\) where \(0 = x_0 < x_1 < \cdots < x_n = b\). Pick \(y \in \oointv{x_{k - 1}}{x_k}\) for some \(k\).

    Set
    \[
        \partition' = \partition \cup \brce{y} = \brce{x_0, \ldots, x_{k - 1}, y, x_k, \ldots, x_n}.
    \]

    Then, the only change in the sum is the supremum
    \[
        \sup_{x \in \ccintv{x_{n - 1}}{y}} f\para{x}, \sup_{x \in \ccintv{y}{x_n}} f\para{x} \leq \sup_{x \in \ccintv{x_{n - 1}}{x_n}} f\para{x}.
    \]

    Multiplying by the widths, we can see that
    \[
        \para{y - x_{k - 1}} \sup_{x \in \ccintv{x_{k - 1}}{y}} f\para{x} + \para{x_k - y} \sup_{x \in \ccintv{y}{x_k}} f\para{x} \leq \para{x_k - x_{k - 1}} \sup_{x \in \ccintv{x_{k - 1}}{x_k}} f\para{x}.
    \]

    Hence, we have
    \[
        \upperSum\para{f, \partition'} \leq \upperSum \para{f, \partition}.
    \]

    Similarly, we may show (replacing with infimums and the opposite inequalities), that
    \(
        \lowerSum\para{f, \partition'} \geq \lowerSum \para{f, \partition}.
    \)

    If \(\partition'\) differs by \(n\) points, we repeat this argument \(n\) times (since partitions are defined to be finite).
\end{proof}

\begin{lemma}[Upper Sums are always at least Lower Sums]
    \label{lem:upper-greater-lower-sum}%
    Let \(\partition\) and \(\partition'\) be partitioned \(\ccintv{a}{b}\) (but not necessarily one being the refinement of the other). Then we have
    \[
        \lowerSum\para{f, \partition} \leq \upperSum\para{f, \partition'}
    \]
    and similarly
    \[
        \lowerSum\para{f, \partition'} \leq \upperSum\para{f, \partition}.
    \]
\end{lemma}

\begin{proof}
    We take the common refinement \(\partition \cup \partition'\). Therefore, by \autoref{lem:refinement-better-estimate},
    \[
        \lowerSum\para{f, \partition} \leq \upperSum\para{f, \partition \cup \partition'} \leq \upperSum\para{f, \partition \cup \partition'} \leq \upperSum\para{f, \partition'}.
    \]
\end{proof}

\begin{corollary}[Upper Integrals are always at least Lower Integraks]
    \label{cor:upper-greater-lower-int}%
    We have for any bounded \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\), that
    \[
        \lowerI\para{f} = \lowerInt_{a}^{b} f\para{x} \Diff x \leq \upperInt_{a}^{b} f\para{x} \Diff x = \upperI \para{f}.
    \]
\end{corollary}

\begin{proof}
    We have \(\lowerSum\para{f, \partition} \leq \upperSum\para{f, \partition'}\) from \autoref{lem:upper-greater-lower-sum}. Taking the infimum over all \(\partition'\) gives
    \[
        \lowerSum\para{f, \partition} \leq \inf_{\partition'} \upperSum\para{f, \partition'} = \upperI.
    \]

    Taking the supremum over all \(\partition\) gives
    \[
        \lowerI = \sup_{\partition} \lowerSum\para{f, \partition} \leq \upperI
    \]
    as desired.
\end{proof}

Another way of viewing integrability is as follows: we can always find a partition \(\partition\) of \(\ccintv{a}{b}\), such that the upper sum (overestimate) and the lower sum (underestimate) differ by, at most, any arbitrary threshold that we specify.

\begin{proposition}[Riemann Integrability Criteria]
    \label{prop:riemann-integrability-criteria}%
    Let \(\functiondomain{f}{\ccintv{a}{b}}\) be bounded. Then \(f\) is Riemann integrable, if and only if
    \[
        \forall \epsilon > 0, \exists \partition: \upperSum\para{f, \partition} - \lowerSum\para{f, \partition} < \epsilon,
    \]
    where \(\partition\) is a partition.
\end{proposition}

\begin{proof}
    Suppose that the criteria is true. We have
    \[
        0 \leq \upperI\para{f} - \lowerI\para{f} \leq \upperSum\para{f, \partition} - \lowerSum\para{f, \partition} < \epsilon,
    \]
    for any \(\epsilon > 0\), and so we must have \(\upperI\para{f} - \lowerI\para{f} = 0\).

    Now, suppose that \(f\) is Riemann integrable. By the definition of the supremum and infimum, we can always find partitions \(\partition, \partition'\) such that
    \[
        \upperSum\para{f, \partition} \leq \upperI\para{f} + \frac{\epsilon}{2}
    \]
    and
    \[
        \lowerSum\para{f, \partition'} \geq \lowerI\para{f} + \frac{\epsilon}{2}.
    \]

    Therefore,
    \[
        \upperSum\para{f, \partition} - \lowerSum\para{f, \partition'} \leq \epsilon.
    \]

    To conclude, we take a common refinement \(\partition \cup \partition'\), and then we have
    \[
        \upperSum\para{f, \partition \cup \partition'} - \lowerSum\para{f, \partition \cup \partition'} \leq \upperSum\para{f, \partition} - \lowerSum\para{f, \partition'} \leq \epsilon,
    \]
    so we are done.
\end{proof}

Note that this is equivalent to the following sequential version.

\begin{proposition}[Sequential Riemann Integrability Criteria]
    \label{prop:sequential-riemann-integrability-criteria}%
    Let \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) be bounded. Then \(f\) is Riemann integrable, if and only if there is a sequence of partitions \(\para{\partition_n}_{n \in \NN}\), such that
    \[
        \upperSum\para{f, \partition_n} - \lowerSum\para{f, \partition_n} \to 0
    \]
    as \(n \to \infty\).
\end{proposition}

\begin{examples}[Integrals of Functions]
    \label{ex:integral-fn}%
    \begin{enumerate}
        \item Let \(f\para{x} = x\) be defined on \(\ccintv{0}{1}\). Let the partition be
        \[
            \partition_n = \brce{0, \frac{1}{n}, \ldots, \frac{n - 1}{n}, 1}.
        \]

        Then, we have
        \begin{align*}
            \upperSum\para{f, \partition_n} &= \sum_{j = 1}^{n} \frac{1}{n} \sup_{x \in \ccintv{\frac{j - 1}{n}}{\frac{j}{n}}} x \\
            &= \frac{1}{n^2} \cdot \sum_{j = 1}^{n} \sup_{x \in \ccintv{j - 1}{j}} x \\
            &= \frac{1}{n^2} \sum_{j = 1}^{n} j \\
            &= \frac{1}{n^2} \cdot \frac{n \para{n + 1}}{2} \\
            &= \frac{1}{2} \para{1 + \frac{1}{n}}.
        \end{align*}

        Similarly, we compute the lower sum as
        \begin{align*}
            \lowerSum\para{f, \partition_n} &= \sum_{j = 1}^{n} \frac{1}{n}\inf_{x \in \ccintv{\frac{j - 1}{n}}{\frac{j}{n}}} x \\
            &= \frac{1}{n^2} \cdot \frac{n \para{n - 1}}{2} \\
            &= \frac{1}{2} \para{1 - \frac{1}{n}}.
        \end{align*}

        With the original definition in \autoref{def:riemann-integral}, we notice
        \[
            \lowerI\para{f} = \sup_{\partition} \lowerSum\para{f, \partition} \geq \sup_{n \in \NN} \lowerSum\para{f, \partition_n} = \frac{1}{2},
        \]
        and similarly,
        \[
            \upperI\para{f} = \inf_{\partition} \upperSum\para{f, \partition} \leq \inf_{n \in \NN} \upperSum\para{f, \partition_n} = \frac{1}{2},
        \]

        But by \autoref{cor:upper-greater-lower-int}, we have
        \[
            \frac{1}{2} \leq \lowerI\para{f} \leq \upperI\para{f} \leq \frac{1}{2}
        \]
        and hence
        \[
            \lowerI\para{f} = \upperI\para{f} = \frac{1}{2}
        \]
        so \(\int_{0}^{1} f\para{x} \Diff x = \frac{1}{2}\).

        With \autoref{prop:sequential-riemann-integrability-criteria} (Sequential Riemann Integrability Criteria), we note
        \[
            \upperSum\para{f, \partition_{n}} - \lowerSum\para{f, \partition_{n}} = \frac{1}{n} \to 0
        \]
        as \(n \to \infty\). So this function is integrable.

        \item Consider the piecewise-defined function
        \[
            f\para{x} = \begin{cases}
                0, & x \leq \frac{1}{2}, \\
                1, & x > \frac{1}{2}
            \end{cases}
        \]
        defined on \(\ccintv{0}{1}\).

        Similarly, we take the same partitions \(\partition_n\). Then
        \[
            \sup_{x \in \ccintv{x_{k - 1}}{x_k}} f\para{x} - \inf_{x \in \ccintv{x_{k - 1}}{x_k}} f\para{x} = \begin{cases}
                1, & x_{k - 1} \leq \frac{1}{2} < x_k,\\
                0, & \text{otherwise}.
            \end{cases}
        \]

        There is always precisely one interval satisfying this difference being \(1\), so
        \[
            \upperSum\para{f, \partition_n} - \lowerSum\para{f, \partition_n} = \frac{1}{n} \to 0
        \]
        as \(n \to \infty\). So the function is integrable.
    \end{enumerate}
\end{examples}

\subsubsection{Sufficient Conditions on Integrability}

We want to find some more sufficient conditions on integrability.

\begin{proposition}[Monotone Functions are Integrable]
    \label{prop:monotonic-integrable}%
    Any monotonic \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) is integrable.
\end{proposition}

\begin{proof}
    First, \(f\) is bounded by \(\max\brce{\abs{f\para{a}}, \abs{f\para{b}}}\) so is indeed bounded.

    Note that for any partition \(\partition\), we have
    \[
        \upperSum\para{f, \partition} - \lowerSum\para{f, \partition} = \sum_{j = 1}^{n} \para{\sup_{I_j} f - \inf_{I_j} f} \abs{I_j},
    \]
    for \(I_j = \ccintv{x_{j - 1}}{x_j}\), and \(\abs{I_j} = x_j - x_{j - 1}\).

    But since \(f\) is monotone, the supremum and the infimums are obtained at the endpoints of the intervals. Without loss of generality, suppose \(f\) is increasing, and
    \[
        \sup_{x \in I_j} f\para{x} = f\para{x_j}, \quad \text{and} \quad \inf_{x \in I_j} f\para{x} = f\para{x_{j - 1}}.
    \]

    Hence, we have
    \[
        \upperSum\para{f, \partition} - \lowerSum\para{f, \partition} = \sum_{j = 1}^{n} \brac{f\para{x_j} - f\para{x_{j - 1}}} \abs{I_j}.
    \]

    We set a particular partition
    \[
        \partition_n = \brce{a, a + \frac{b - a}{n}, a + \frac{2\para{b - a}}{n}, \ldots, a + \frac{\para{n - 1} \para{n - a}}{n}, b},
    \]
    and so \(\abs{I_j} = \frac{b - a}{n}\). Hence,
    \[
        \upperSum\para{f, \partition_n} - \lowerSum\para{f, \partition_n} = \frac{b - a}{n} \cdot \sum_{j = 1}^{n} \brac{f\para{a + \frac{j \para{b - a}}{n}} - f\para{a + \frac{\para{j - 1} \para{b - a}}{n}}}
    \]
    and by telescoping, we have
    \[
        \upperSum\para{f, \partition_n} - \lowerSum\para{f, \partition_n} = \frac{b - a}{n} \brac{f\para{b} - f\para{a}} \to 0
    \]
    as \(n \to \infty\). Using \autoref{prop:sequential-riemann-integrability-criteria} (Sequential Version of Riemann Integrability Criteria), \(f\) is integrable, as desired.
\end{proof}

\begin{proposition}[Continuous Functions are Integrable]
    \label{prop:continuous-integrable}%
    Any continuous \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) is integrable.
\end{proposition}

\begin{proof}
    \(f\) is bounded by \autoref{thm:extreme-value} (Extreme Value Theorem).

    We show by contradiction.

    If \(f\) is not integrable, by \autoref{prop:riemann-integrability-criteria} (Riemann Integrability Criteria), there is some \(\epsilon > 0\), such that
    \[
        \forall \partition: \upperSum\para{f, \partition} - \lowerSum\para{f, \partition} > \epsilon.
    \]

    Therefore, we have
    \[
        \para{b - a} \cdot \max_{1 \leq j \leq n} \osc_{I_j} f \geq \sum_{k = 1}^{n} \abs{I_k} \cdot \para{\sup_{I_j} f - \inf_{I_j} f} = \upperSum\para{f, \partition} - \lowerSum\para{f, \partition} > \epsilon,
    \]
    where we define
    \[
        \osc f = \sup f - \inf f
    \]
    to be the \emph{oscillation}.

    Therefore, there is some \(j\) in any partition \(\partition\), such that
    \[
        \osc_{x \in I_j} f\para{x} = \sup_{x_1 \in I_j} f\para{x_1} - \inf_{x_2 \in I_j} f\para{x_2} > \frac{\epsilon}{b - a}.
    \]

    Therefore, by \autoref{thm:extreme-value} (Extreme Value Theorem), there is some \(y, z \in \ccintv{x_{j - 1}}{x_j}\), such that
    \[
        f\para{y} - f\para{z} > \frac{\epsilon}{b - a}.
    \]

    Since this is true for all partitions, it should also be true for the \(\partition_n\)s given in the previous proof (by splitting up equally). So for each \(n\), we have some \(y_n, z_n\) such that
    \[
        \abs{y_n - z_n} < \frac{b - a}{n} \quad \text{and} \quad \abs{f\para{y_n} - f\para{z_n}} > \frac{\epsilon}{b - a}.
    \]

    By \autoref{thm:bolzano-weierstrass} (Bolzano--Weierstrass Theorem), \(\para{y_n}, \para{z_n}\) have convergent subsequences. So for the subsequence index \(\para{n_k}\), we have \(\para{y_{n_k}} \to y\) for some \(y \in \ccintv{a}{b}\), and hence
    \[
        \abs{z_{n_k} - y_{n_k}} < \frac{b - a}{n_k} \to 0
    \]
    as \(n \to \infty\), so \(z_{n_k} \to y\) as well. Imposing sequential continuity, but \(\para{f\para{y_{n_k}}}\) and \(\para{f\para{z_{n_k}}}\) do not have the same limit. This gives a contradiction with the continuity of \(f\).
\end{proof}

\begin{definition}[Piecewise Continuous]
    \label{def:piecewise-continuous}%
    Let \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\). We say \(f\) is \emph{piecewise continuous} if there is some partition \(\partition\) of \(\ccintv{a}{b}\), such that the restriction on every open interval
    \[
        \rightbar{f}_{\oointv{x_{j - 1}}{x_j}}
    \]
    is continuous, and has a finite (one-sided) limit as \(x \to x_j\).
\end{definition}

\begin{proposition}[Piecewise Continuous Functions are Integrable]
    \label{prop:piecewise-continuous-integrable}%
    Any piecewise continuous function \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) is integrable, and
    \[
        \int_{a}^{b} f\para{x} \Diff x = \sum_{j = 1}^{k} \int_{x_{j - 1}}^{x_j} \overline{f_j} \para{x} \Diff x,
    \]
    where \(\overline{f_j}\)s are defined to be
    \[
        \overline{f_j} \para{x} = \begin{cases}
            f\para{x}, & x_{j - 1} < x < x_j, \\
            f\para{x_j^{-}} = \lim_{x \to x_j^{-}} f\para{x}, & x = x_j, \\
            f\para{x_{j - 1}^{+}} = \lim_{x \to x_{j - 1}^{+}} f\para{x}, & x = x_{j - 1}.
        \end{cases}
    \]
\end{proposition}

This is a very strong result. It not only guarantees integrability, but also tells us a way to calculate the integral.

\begin{lemma}[Altering Finitely Many Points retains the Integral]
    \label{lem:individual-points-retains-integrable}%
    Suppose \(\functiondomain{f, g}{\ccintv{a}{b}}{\RR}\) is bounded. Suppose \(g\) is integrable, and the set of difference points
    \[
        \set{x \in \ccintv{a}{b}}{f\para{x} \neq g\para{x}} = \brce{z_1, \ldots, z_N}
    \]
    for some finite \(N\). Then \(f\) is integrable, and
    \[
        \int_{a}^{b} f\para{x} \Diff x = \int_{a}^{b} g\para{x} \Diff x.
    \]
\end{lemma}

\begin{proof}
    Set \(M = \sup_{x \in \ccintv{a}{b}} \abs{f\para{x}}\). Fix \(\epsilon > 0\), then by \autoref{prop:riemann-integrability-criteria} (Riemann Integrability Criteria), there is some \(\partition\) of \(\ccintv{a}{b}\) such that
    \[
        \upperSum\para{g, \partition} - \lowerSum\para{g, \partition} < \epsilon.
    \]

    We choose the intervals \(J_k = \ccintv{z_k - r_k}{z_k + r_k}\) with
    \[
        \sum_{k = 1}^{N} \abs{J_k} = 2 \sum_{k = 1}^{N} r_k < \epsilon_0
    \]
    and such that they are disjoint. (If \(z_k = a\), we take the interval \(\ccintv{z_k}{z_k + r_k}\) and similarly for \(z_k = b\).)

    Consider the new partition
    \[
        \partition' = \partition \cup \brce{z_1 \pm r_1, z_2 \pm r_2, \ldots, z_N \pm r_N}.
    \]

    Let
    \[
        J = \bigcup_{k = 1}^{N} J_k.
    \]

    Note that
    \[
        \upperSum\para{g, \partition'} - \lowerSum\para{g, \partition'} \leq \upperSum\para{g, \partition} - \lowerSum\para{g, \partition} < \epsilon.
    \]

    We now bound the upper and lower sums of \(f\). We have
    \begin{align*}
        \upperSum\para{f, \partition'} - \lowerSum\para{f, \partition'} &= \sum_{j = 1}^{n} \abs{I_j} \para{\sup_{I_j} - \inf_{I_j} f} \\
        &= \sum_{j = 1, I_j \subset J}^{n} \abs{I_j} \para{\sup_{I_j} - \inf_{I_j} f} + \sum_{j = 1, I_j \nsubset J}^{n} \abs{I_j} \para{\sup_{I_j} - \inf_{I_j} f}.
    \end{align*}

    Note that \(I_j \nsubset J\), then \(f = g\), and hence
    \begin{align*}
        \sum_{j = 1, I_j \nsubset J}^{n} \abs{I_j} \para{\sup_{I_j} - \inf_{I_j} f} &\leq \sum_{j = 1}^{n} \abs{I_j} \para{\sup_{i_j} g - \inf_{I_j} g} \\
        &= \upperSum\para{g, \partition'} - \lowerSum\para{g, \partition'} \\
        &< \epsilon.
    \end{align*}

    For \(I_j \subset J\), then we have
    \[
        \para{\sup_{I_j} f - \inf_{I_j} f} \leq 2M,
    \]
    and hence
    \[
        \sum_{j = 1, I_j \subset J}^{n} \abs{I_j} \para{\sup_{I_j} f - \inf_{I_j} f} \leq 2M \sum_{j = 1, I_j \subset J}^{n} \abs{I_j} \leq 2M \cdot \sum_{k = 1}^{N} \abs{J_k} < 2M \epsilon.
    \]

    Hence,
    \[
        \upperSum\para{f, \partition'} - \lowerSum\para{f, \partition'} < \para{2M + 1} \epsilon.
    \]
\end{proof}

Now we prove \autoref{prop:piecewise-continuous-integrable} (Piecewise Continuous Functions are Integrable).

\begin{proof}[of \autoref{prop:piecewise-continuous-integrable} (Piecewise Continuous Functions are Integrable)]
    By \autoref{lem:individual-points-retains-integrable}, we have that the restrictions \(\rightbar{f}_{\ccintv{x_{j - 1}}{x_j}}\) is integrable, and
    \[
        \int_{x_{j - 1}}^{x_j} f\para{x} \Diff x = \int_{x_{j - 1}}^{x_j} \overline{f_j} \para{x} \Diff x.
    \]

    To conclude, we use \autoref{prop:additivity-integral} (Additivity of the Integral).

    To calculate the integrals, we have
    \[
        \lowerSum\para{f, \partition'} \leq \int_{a}^{b} f\para{x} \Diff x \leq \upperSum\para{f, \partition'} \text{ and } \lowerSum\para{f, \partition'} \leq \int_{a}^{b} g\para{x} \Diff x \leq \upperSum\para{g, \partition'}.
    \]

    Hence,
    \begin{align*}
        \int_{a}^{b} f\para{x} \Diff x - \int_{a}^{b} g\para{x} \Diff x &\leq \upperSum\para{f, \partition'} - \upperSum\para{g, \partition'} + \upperSum\para{g, \partition'} - \lowerSum\para{g, \partition'} \\
        &< \epsilon + \epsilon.
    \end{align*}
\end{proof}

\begin{proposition}[Additivity of Integral]
    \label{prop:additivity-integral}%
    Let \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) and \(c \in \oointv{a}{b}\). Then \(f\) is integrable, if and only if the restrictions \(\rightbar{f}_{\ccintv{a}{c}}\) and \(\rightbar{f}_{\ccintv{c}{b}}\) are integrable, and furthermore
    \[
        \int_{a}^{b} f\para{x} \Diff x = \int_{a}^{c} \rightbar{f}_{\ccintv{a}{c}} \para{x} \Diff x + \int_{c}^{b} \rightbar{f}_{\ccintv{c}{b}} \para{x} \Diff x.
    \]
\end{proposition}

\begin{proof}
    We set \(I_L = \ccintv{a}{c}\), \(I_R = \ccintv{c}{b}\), and let the restrictions be
    \[
        f_L = \rightbar{f}_{I_L}, \quad f_R = \rightbar{f}_{I_R}.
    \]

    If \(f\) is integrable, then there is some partition \(\partition\) of \(\ccintv{a}{b}\), such that
    \[
        \upperSum\para{f, \partition} - \lowerSum\para{f, \partition} < \epsilon.
    \]

    We add the point \(c\) into the partition (and it will only decrease the difference, so does not change the inequality), and let \(c = x_l\) for some \(l \in \brce{0, \ldots, n}\). Then, the partition can be split up into two parts
    \[
        \partition = \partition_L \cup \partition_R,
    \]
    where
    \[
        \partition_L = \brce{a = x_0, x_1, \ldots, x_{l - 1}, x_l = c}, \quad \partition_R = \brce{c = x_l, x_{l + 1}, \ldots, x_{n - 1}, x_n = b},
    \]
    which are partitions of \(I_L = \ccintv{a}{c}\) and \(I_R = \ccintv{c}{b}\) respectively.
    
    Furthermore,
    \[
        \upperSum\para{f, \partition} = \upperSum\para{f_L, \partition_L} + \upperSum\para{f_R, \partition_R}
    \]
    and similarly,
    \[
        \lowerSum\para{f, \partition} = \lowerSum\para{f_L, \partition_L} + \lowerSum\para{f_R, \partition_R}.
    \]

    Summing these two, we have
    \[
        \para{\upperSum\para{f_L, \partition_L} - \lowerSum\para{f_L, \partition_L}} + \para{\upperSum\para{f_R, \partition_R} - \lowerSum\para{f_R, \partition_R}} < \epsilon,
    \]
    and hence both differences are \(< \epsilon\) since they are non-negative. This tells us \(f_L\) and \(f_R\) are both integrable.

    Suppose now that \(f_L\) and \(f_R\) are integrable. Now, suppose \(\partition_L\) and \(\partition_R\) are partitions of \(I_L\) and \(I_R\) respectively, such that
    \[
        \upperSum\para{f_L, \partition_L} - \lowerSum\para{f_L, \partition_L} < \epsilon, \quad \upperSum\para{f_R, \partition_R} - \lowerSum\para{f_R, \partition_R} < \epsilon.
    \]

    The union of the two partitions \(\partition = \partition_L \cup \partition_R\) is a partition of the whole interval \(\ccintv{a}{b}\).

    Therefore,
    \[
        \upperSum\para{f, \partition} - \lowerSum\para{f, \partition} < \epsilon,
    \]
    and hence we may conclude \(f\) is integrable in \(\ccintv{a}{b}\).

    To show the equality of the integrals holds, we have for any \(\epsilon > 0\), that there are partitions \(\partition_L\) and \(\partition_R\), respectively, that
    \[
        \int_{a}^{b} f\para{x} \Diff x \geq \lowerSum\para{f_L, \partition_L} + \lowerSum\para{f_R, \partition_R} \geq \int_{a}^{c} f_L\para{x} \Diff x + \int_{c}^{b} f_R\para{x} \Diff x - 2\epsilon,
    \]
    and similarly,
    \[
        \int_{a}^{b} f\para{x} \Diff x \leq \upperSum\para{f_L, \partition_L} + \upperSum\para{f_R, \partition_R} \leq \int_{a}^{c} f_L\para{x} \Diff x + \int_{c}^{b} f_R\para{x} \Diff x + 2\epsilon.
    \]

    Therefore,
    \[
        \abs{\int_{a}^{b} f\para{x} \Diff x - \para{\int_{a}^{c} f_L\para{x} \Diff x + \int_{c}^{b} f_R\para{x} \Diff x}} < 2 \epsilon.
    \]

    Therefore, the integrals are equal, as desired.
\end{proof}

\autoref{ex:thomae-function} will show a function which is not monotone or piecewise continuous, but is still integrable.

\begin{example}[Thomae's Function / Popcorn Function]
    \label{ex:thomae-function}%
    Thomae's Function is defined as
    \[
        \function{f}{\ccintv{0}{1}}{\RR}{x}{\begin{cases}
            \frac{1}{q}, & x \in \QQ, x = \frac{p}{q}, \gcd\brce{p, q} = 1, \\
            0, & x \notin \QQ.
        \end{cases}}
    \]

    Since \(\RR \setminus \QQ\) is dense in \(\RR\), we have
    \[
        \lowerSum\para{f, \partition} = 0
    \]
    for any partition \(\partition\) of \(\ccintv{0}{1}\), and hence \(\lowerI\para{f} = 0\).

    We claim \(f\) is integrable, so we want to show that for any \(\epsilon > 0\), there is some partition \(\partition\) such that \(\upperSum\para{f, \partition} < \epsilon\).

    We pick \(N \in \NN\), such that \(\frac{1}{N} < \epsilon\). Consider the set of points
    \[
        X_N = \set{x \in \ccintv{0}{1}}{f\para{x} \geq \frac{1}{N}}.
    \]

    Note that this set is finite, since
    \[
        X_N \subseteq \set{\frac{p}{q}}{1 \leq q \leq N, 0 \leq p \leq q}.
    \]

    Therefore, we let
    \[
        X_N = \brce{y_1, \ldots, y_M}
    \]
    for some finite \(M \in \NN\). Now, define a partition \(\partition\), such that:
    \begin{enumerate}
        \item each \(y_k\) is in some subinterval of \(P\); and
        \item these subintervals have length \(< \frac{\epsilon}{M}\).
    \end{enumerate}

    Computing the upper Riemann sum, we have
    \begin{align*}
        \upperSum\para{f, \partition} &= \sum_{I \in \partition} \abs{I} \cdot \sup_{x \in I} f\para{x} \\
        &= \sum_{I \in \partition, I \cap X_N \neq \emptyset} \abs{I} \cdot \sup_{x \in I} f\para{x} + \sum_{I \in \partition, I \cap X_N \neq \emptyset} \abs{I} \cdot \sup_{x \in I} f\para{x} \\
        &\leq \sum_{I \in \partition, I \cap X_N \neq \emptyset} \abs{I} + \frac{1}{N} \cdot \sum_{I \in \partition, I \cap X_N = \emptyset} \abs{I} \\
        &\leq M \cdot \frac{\epsilon}{M} + \frac{1}{N} \cdot 1 \\
        &< 2 \epsilon.
    \end{align*}

    Comparing to the \autoref{ex:non-riemann-integrable-fn} (Dirichlet Function), these function may both look `crazy' -- yet Riemann integrability properties differ -- but there are reasons for that.
\end{example}

\begin{proposition}[Countable Discontinuities implies Integrability]
    \label{prop:countable-discontinuity-implies-integrability}%
    Let \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) be bounded, and let its set of discontinuities be
    \[
        D = \set{x \in \ccintv{a}{b}}{f \text{ is not continuous at }x}.
    \]
    
    If \(D\) is finite or countable, then \(f\) is Riemann integrable.
    
    \textit{[The countably infinite case is non-examinable.]}
\end{proposition}

\begin{proof}[of Finite Case]
    Without loss of generality, let the set of discontinuities be \(D = \brce{d_1 < \cdots < d_k}\). We consider the restrictions
    \[
        \rightbar{f}_{\ccintv{a}{d_1}}, \rightbar{f}_{\ccintv{d_1}{d_2}}, \ldots, \rightbar{f}_{\ccintv{d_{k - 1}}{d_k}}, \rightbar{f}_{\ccintv{d_k}{b}}.
    \]

    It suffices to show that all these functions are integrable, since if so, we would have
    \begin{align*}
        &\phantom{=} \int_{a}^{d_1} \rightbar{f}_{\ccintv{a}{d_1}} \para{x} \Diff x + \int_{d_1}^{d_2} \rightbar{f}_{\ccintv{d_1}{d_2}} \para{x} \Diff x + \cdots + \int_{d_{k - 1}}^{d_k} \rightbar{f}_{\ccintv{d_{k - 1}}{d_k}} \para{x} \Diff x + \int_{d_k}^{b} \rightbar{f}_{\ccintv{d_k}{b}} \para{x} \Diff x \\
        &= \int_{a}^{d_1} f\para{x} \Diff x + \int_{d_1}^{d_2} f\para{x} \Diff x + \cdots + \int_{d_{k - 1}}^{d_k} f\para{x} \Diff x + \int_{d_k}^{b} f\para{x} \Diff x \\
        &= \int_{a}^{b} f\para{x} \Diff x.
    \end{align*}

    Let \(\functiondomain{g}{\ccintv{c}{d}}{\RR}\) be one of the restrictions above, which may only be discontinuous at \(c\) or \(d\). Since \(f\) is bounded, we have \(g\) is bounded, so let \(M = \sup_{x \in \ccintv{c}{d}} \abs{g\para{x}}\). Then for any subset \(S \subseteq \ccintv{c}{d}\), we have
    \[
        \osc_{x \in S} f\para{x} \leq 2M.
    \]

    Let \(\epsilon > 0\) be arbitrary. Take \(\delta < \min\brce{\frac{d - c}{2}, \frac{\epsilon}{6M}}\), and the restriction of \(g\) on \(\ccintv{c + \delta}{d - \delta}\). \(g\) is continuous on this interval, and hence integrable. Let \(\partition'\) be a partition on \(\ccintv{c + \delta}{d - \delta}\), such that
    \[
        \upperSum\para{g, \partition'} - \lowerSum\para{g, \partition'} < \frac{\epsilon}{3}.
    \]

    Now, we consider the partition \(\partition = \brce{c, d} \cup \partition'\). Then, we have
    \begin{align*}
        \upperSum\para{g, \partition} - \lowerSum\para{g, \partition} &= \delta \cdot \osc_{x \in \ccintv{c}{c + \delta}} f\para{x} + \delta \cdot \osc_{x \in \ccintv{d - \delta}{d}} f\para{x} + \para{\upperSum\para{g, \partition'} - \lowerSum\para{g, \partition'}} \\
        &< \frac{\epsilon}{6M} \cdot 2M + \frac{\epsilon}{6M} \cdot 2M + \frac{\epsilon}{3} \\
        &= \epsilon.
    \end{align*}

    This shows \(g\) is integrable, and hence by the above argument showing \(f\) is integrable as well.
\end{proof}

\begin{remark}
    If \(f\) is not Riemann integrable, then \(D\) must be uncountable. However, there are functions with \(D\) uncountable, yet \(f\) is Riemann integrable. One example of this is the indicator of the Cantor Set. (Since the Cantor Set has zero Lebesgue Measure.)
\end{remark}

\subsubsection{Properties of Riemann Sums and Integrals}

We finally prove several fundamental properties of integrals. However, before this, we first write down lemmas on supremum/infimums and upper/lower sums.

\begin{lemma}[Properties of Supremum and Infimum]
    \label{lem:properties-supremum-infimum}%
    Let \(I\) be a closed and bounded interval, and \(\functiondomain{f, g}{I}{\RR}\) be bounded.

    \begin{enumerate}
        \item \textit{Supremums and Infimums preserve Weak Inequalities.} If \(f\para{x} \leq g\para{x}\) for all \(x\), then
        \[
            \sup_{x \in I} f\para{x} \leq \sup_{x \in I} g\para{x}
            \quad \text{and} \quad
            \inf_{x \in I} f\para{x} \leq \inf_{x \in I} g\para{x}.
        \]
        \item \textit{Supremum of the Negative is Negative the Infimum.} We have
        \[
            \sup_{x \in I} \para{-f\para{x}} = -\inf_{x \in I} f\para{x}
            \quad \text{and} \quad
            \inf_{x \in I} \para{-f\para{x}} = -\sup_{x \in I} f\para{x}.
        \]

        \item \textit{Supremums and Infimums preserve Scalar Multiples.} For any \(\lambda > 0\), we have
        \[
            \sup_{x \in I} \para{\lambda f\para{x}} = \lambda \sup_{x \in I} f\para{x}
            \quad \text{and} \quad
            \inf_{x \in I} \para{\lambda f\para{x}} = \lambda \inf_{x \in I} f\para{x}
        \]

        \item \textit{Triangular Inequalities.} We have
        \[
            \sup_{x \in I} \para{f\para{x} + g\para{x}} \leq \sup_{x \in I} f\para{x} + \sup_{x \in I} g\para{x}
            \quad \text{and} \quad
            \inf_{x \in I} \para{f\para{x} + g\para{x}} \geq \inf_{x \in I} f\para{x} + \inf_{x \in I} g\para{x}.
        \]
        
        \item \textit{Absolute Values.} We have
        \[
            \sup_{x \in I} \abs{f\para{x}} - \inf_{x \in I} \abs{f\para{x}} \leq \sup_{x \in I} f\para{x} - \inf_{x \in I} f\para{x}.
        \]

        \item \textit{Squares.} We have
        \[
            \sup_{x \in I} \para{f\para{x}^2} - \inf_{x \in I} \para{f\para{x}^2} \leq 2 \sup_{x \in I} \abs{f\para{x}} \para{\sup_{x \in I} f\para{x} - \inf_{x \in I} f\para{x}}.
        \]
    \end{enumerate}
\end{lemma}

\begin{proof}
    \begin{enumerate}
        \item We prove for the supremum. We have for any \(x \in I\), that
        \[
            \sup_{x \in I} g\para{x} \geq g\para{x} \geq f\para{x},
        \]
        and hence
        \[
            \sup_{x \in I} f\para{x} \leq \sup_{x \in I} g\para{x}.
        \]

        \item We prove for the supremum. On one hand, we have
        \begin{align*}
            &\phantom{\implies} \forall x \in I: \inf_{x \in I} f\para{x} \leq f\para{x} \\
            &\implies \forall x \in I: - \inf_{x \in I} f\para{x} \geq -f\para{x} \\
            &\implies - \inf_{x \in I} f\para{x} \geq \sup_{x \in I} \para{- f\para{x}}.
        \end{align*}

        On the other hand, we have
        \begin{align*}
            &\phantom{\implies} \forall x \in I: \sup_{x \in I} \para{- f\para{x}} \geq -f\para{x} \\
            &\implies \forall x \in I: f\para{x} \geq -\sup_{x \in I} \para{- f\para{x}}  \\
            &\implies \inf f\para{x} \geq - \sup_{x \in I} \para{- f\para{x}} \\
            &\implies - \inf f\para{x} \leq \sup_{x \in I} \para{- f\para{x}}.
        \end{align*}

        \item We prove for the supremum. On one hand, we have
        \begin{align*}
            &\phantom{\implies} \forall x \in I: \sup_{x \in I} \para{\lambda f\para{x}} \geq \lambda f\para{x} \\
            &\implies \forall x \in I: \frac{1}{\lambda} \cdot \sup_{x \in I} \para{\lambda f\para{x}} \geq f\para{x} \\
            &\implies \frac{1}{\lambda} \cdot \sup_{x \in I} \para{\lambda f\para{x}} \geq \sup_{x \in I} f\para{x} \\
            &\implies \sup_{x \in I} \para{\lambda f\para{x}} \geq \lambda \sup_{x \in I} f\para{x}.
        \end{align*}

        On the other hand, we have
        \begin{align*}
            &\phantom{\implies} \forall x \in I: \sup_{x \in I} \para{f\para{x}} \geq f\para{x} \\
            &\implies \forall x \in I: \lambda \sup_{x \in I} \para{f\para{x}} \geq \lambda f\para{x} \\
            &\implies \lambda \sup_{x \in I} \para{f\para{x}} \geq \sup_{x \in I} \para{\lambda f\para{x}}.
        \end{align*}

        \item We prove for the supremum. We have for any \(x \in I\), that
        \[
            \sup_{x \in I} f\para{x} \geq f\para{x}
            \quad \text{and} \quad
            \sup_{x \in I} g\para{x} \geq g\para{x}.
        \]

        Summing them, we have
        \[
            \sup_{x \in I} f\para{x} + \sup_{x \in I} g\para{x} \geq f\para{x} + g\para{x},
        \]
        and hence
        \[
             \sup_{x \in I} f\para{x} + \sup_{x \in I} g\para{x} \geq \sup_{x \in I} \para{f\para{x} + g\para{x}}.
        \]

        \item If \(f \geq 0\) on all of \(I\), or \(f \leq 0\) on all of \(I\), this is an immediate consequence of (2).
        
        If \(\inf_{x \in I} f\para{x} \leq 0 \leq \sup_{x \in I} f\para{x}\), then
        \begin{align*}
            \sup_{x \in I} \abs{f\para{x}} - \inf_{x \in I} \abs{f\para{x}} &\leq \sup_{x \in I} \abs{f \para{x}} \\
            &\leq \max \brce{\sup_{x \in I} f\para{x}, \sup_{x \in I} \para{-f\para{x}}} \\
            &\leq \sup_{x \in I} f\para{x} + \sup_{x \in I} \para{-f\para{x}} \\
            &= \sup_{x \in I} f\para{x} - \inf_{x \in I} f\para{x}.
        \end{align*}

        \item We have
        \[
            f\para{x}^2 - f\para{y}^2 = \para{f\para{x} - f\para{y}} \para{f\para{x} + f\para{y}}
        \]
        for all \(x, y \in I\). Therefore,
        \begin{align*}
            \sup_{x \in I} f\para{x}^2 - \inf_{y \in I} f\para{y}^2 &\leq \sup_{x \in I} \inf_{y \in I} \para{\brac{f\para{x} - f\para{y}} \brac{f\para{x} + f\para{y}}} \\
            &\leq 2 \sup_{x \in I} \abs{f\para{x}} \sup_{x \in I} \inf_{y \in I} \brac{f\para{x} - f\para{y}} \\
            &= 2 \sup_{x \in I} \abs{f\para{x}} \para{\sup_{x \in I} f\para{x} - \inf_{x \in I} f\para{x}}.
        \end{align*}
    \end{enumerate}
\end{proof}

\begin{lemma}[Properties of Upper and Lower Sum]
    \label{lem:properties-upper-lower-sum}%
    Let \(\functiondomain{f, g}{I}{\RR}\) be bounded, and let \(\partition\) be a partition of \(\ccintv{a}{b}\).

    \begin{enumerate}
        \item \textit{Upper and Lower Sums preserve Weak Inequalities.} If \(f\para{x} \leq g\para{x}\) for all \(x\), then
        \[
            \upperSum\para{f, \partition} \leq \upperSum\para{g, \partition}
            \quad \text{and} \quad
            \lowerSum\para{f, \partition} \leq \lowerSum\para{g, \partition}.
        \]

        \item \textit{Upper Sum of the Negative is Negative the Lower Sum.} We have
        \[
            \upperSum\para{-f, \partition} = -\lowerSum\para{f, \partition}
            \quad \text{and} \quad
            \lowerSum\para{-f, \partition} = -\upperSum\para{f, \partition}.
        \]

        \item \textit{Upper and Lower Sums preserve Scalar Multiples}. For any \(\lambda > 0\), we have
        \[
            \upperSum\para{\lambda f, \partition} = \lambda \upperSum\para{f, \partition}
            \quad \text{and} \quad
            \lowerSum\para{\lambda f, \partition} = \lambda \lowerSum\para{f, \partition}.
        \]

        \item \textit{Triangular Inequalities.} We have
        \[
            \upperSum\para{f + g, \partition} \leq \upperSum\para{f, \partition} + \upperSum\para{g, \partition}
            \quad \text{and} \quad
            \lowerSum\para{f + g, \partition} \geq \lowerSum\para{f, \partition} + \lowerSum\para{g, \partition}.
        \]

        \item \textit{Absolute Values.} We have
        \[
            \upperSum\para{\abs{f}, \partition} - \lowerSum\para{\abs{f}, \partition} \leq \upperSum\para{f, \partition} - \lowerSum\para{f, \partition}.
        \]

        \item \textit{Squares.} We have
        \[
            \upperSum\para{f^2, \partition} - \lowerSum\para{f^2, \partition} \leq 2 \sup_{x \in I} \abs{f\para{x}} \para{\upperSum\para{f, \partition} - \lowerSum\para{f, \partition}}.
        \]
    \end{enumerate}
\end{lemma}

\begin{proof}
    These are direct consequences of application of \autoref{lem:properties-supremum-infimum}.
\end{proof}

\begin{lemma}[Integrals preserve Weak Inequalities]
    \label{lem:integral-preserve-weak-inequalities}%
    Let \(\functiondomain{f, g}{\ccintv{a}{b}}{\RR}\) be integrable. If \(f \leq g\), then
    \[
        \int_{a}^{b} f\para{x} \Diff x \leq \int_{a}^{b} g\para{x} \Diff x.
    \]
\end{lemma}

\begin{proof}
    From \autoref{lem:properties-upper-lower-sum}, we have \(\upperSum\para{f, \partition} \leq \upperSum\para{g, \partition}\), so taking infimum over all partitions gives us the result as desired.
\end{proof}

\begin{lemma}[Integration is Linear]
    \label{lem:integral-linear}%
    Let \(\functiondomain{f, g}{\ccintv{a}{b}}{\RR}\) be integrable. For any \(\lambda \in \RR\), we have
    \[
        \int_{a}^{b} \lambda f\para{x} \Diff x = \lambda \int_{a}^{b} f\para{x} \Diff x,
    \]
    and
    \[
        \int_{a}^{b} \brac{f\para{x} + g\para{x}} \Diff x = \int_{a}^{b} f\para{x} \Diff x + \int_{a}^{b} g\para{x} \Diff x.
    \]
\end{lemma}

\begin{proof}
    Scalar multiplication arises from properties 2 and 3 of \autoref{lem:properties-upper-lower-sum}, dealing with the cases \(\lambda > 0\) and \(\lambda < 0\) (and the trivial case \(\lambda = 0\)) separately.
    
    For the sum, we have from property 4 of \autoref{lem:properties-upper-lower-sum}, that
    \[
        \lowerSum\para{f, \partition} + \lowerSum\para{g, \partition} \leq \lowerSum\para{f + g, \partition} \leq \upperSum\para{f + g, \partition} \leq \upperSum\para{f, \partition} + \upperSum\para{g, \partition}.
    \]

    Therefore,
    \[
        \lowerI\para{f} + \lowerI\para{g} \leq \lowerI\para{f + g} \leq \upperI\para{f + g} \leq \upperI\para{f} + \upperI\para{g},
    \]
    and we have the result as desired.
\end{proof}

\begin{lemma}[Integral of Absolute Value and Triangular Inequality]
    \label{lem:integral-absolute-value-triangular-inequality}%
    Let \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) be integrable. Then \(\abs{f}\) is integrable, and
    \[
        \abs{\int_{a}^{b} f\para{x} \Diff x} \leq \int_{a}^{b} \abs{f\para{x}} \Diff x.
    \]
\end{lemma}

\begin{proof}
    For any partition \(\partition\) of \(\ccintv{a}{b}\), from \autoref{lem:properties-upper-lower-sum}, Property 5, we have
    \[
        \upperSum\para{\abs{f}, \partition} - \lowerSum\para{\abs{f}, \partition} \leq \upperSum\para{f, \partition} - \lowerSum\para{f, \partition},
    \]
    so if \(f \para{x}\) is integrable, we have \(\abs{f \para{x}}\) is integrable.

    Since we have
    \[
        - \abs{f \para{x}} \leq f\para{x} \leq \abs{f\para{x}},
    \]
    by \autoref{lem:integral-preserve-weak-inequalities}, we have
    \[
        -\int_{a}^{b} \abs{f\para{x}} \Diff x \leq \int_{a}^{b} f\para{x} \Diff x \leq \int_{a}^{b} \abs{f\para{x}} \Diff x,
    \]
    and hence
    \[
        \abs{\int_{a}^{b} f\para{x}} \Diff x \leq \int_{a}^{b} \abs{f\para{x}} \Diff x.
    \]
\end{proof}

\begin{lemma}[Product of Integrable Functions is Integrable]
    \label{lem:integral-product}%
    Let \(\functiondomain{f, g}{\ccintv{a}{b}}{\RR}\) be integrable. Then the product \(f\para{x} \cdot g\para{x}\) is integrable.
\end{lemma}

\begin{proof}
    Note that
    \[
        f\para{x} g\para{x} = \frac{1}{4} \brac{\para{f\para{x} + g\para{x}}^2 - \para{f\para{x} - g\para{x}}^2},
    \]
    so it suffices to show that \(f\para{x}\) is integrable implies \(f\para{x}^2\) is integrable.

    For any partition \(\partition\) of \(\ccintv{a}{b}\), we have
    \[
        \upperSum\para{f^2, \partition} - \lowerSum\para{f^2, \partition} \leq 2 \sup_{x \in I} \abs{f\para{x}} \para{\upperSum\para{f, \partition} - \lowerSum\para{f, \partition}}.
    \]

    But note \(2 \sup_{x \in I} \abs{f\para{x}}\) is a constant.
\end{proof}

\begin{remark}
    We do not have a formula for this case, and it is generally the case that
    \[
        \int_{a}^{b} f\para{x} g\para{x} \Diff x \neq \int_{a}^{b} f\para{x} \Diff x \cdot \int_{a}^{b} g\para{x} \Diff x.
    \]
\end{remark}

\subsection{Fundamental Theorem of Calculus}

We will think about the `integral of \(f\)' in this section as a function \(F\), known as the \emph{indefinite integral}, or the \emph{antiderivative} (the name will make sense later), which is the value of the integral of \(f\) over a varying upper bound
\[
    \function{F}{\ccintv{a}{b}}{\RR}{x}{\int_{a}^{x} f\para{t} \Diff t.}
\]

If we want \(F\) to be differentiable, it must be continuous.

\begin{proposition}[Indefinite Integral is Continuous]
    \label{prop:indef-integral-continuous}%
    Let \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) be Riemann integrable, and let \(F\) be defined as above. Then \(F\) is continuous on \(\ccintv{a}{b}\).
\end{proposition}

\begin{proof}
    We have
    \begin{align*}
        \abs{F\para{x + h} - F\para{x}} &= \abs{\int_{a}^{x + h} f\para{t} \Diff t - \int_{a}^{x} f\para{t} \Diff t} \\
        &= \abs{\int_{x}^{x + h} f\para{t} \Diff t} \\
        &\leq \int_{x}^{x + h} \abs{f\para{t}} \Diff t \\
        &\leq \sup_{x \in \ccintv{a}{b}} \abs{f} \cdot \abs{h}.
    \end{align*}

    Hence, as \(h \to 0\), \(\abs{F\para{x + h} - F\para{x}} \to 0\), showing \(F\) is continuous.
\end{proof}

\begin{theorem}[Fundamental Theorem of Calculus, Part I]
    \label{thm:ftc-1}%
    If \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) is Riemann Integrable, and \(f\) is continuous at \(x_0\), then the \(F\) defined above is differentiable at \(x_0\), and
    \[
        F'\para{x_0} = \rightbar{\DiffOp{x}}_{x = x_0} \int_{a}^{x} f\para{t} \Diff t = f\para{x_0}.
    \]
\end{theorem}

\begin{proof}
    We use \autoref{lem:linear-approximation-differentiability}, i.e., we study the function
    \[
        \epsilon\para{h} = \frac{F\para{x_0 + h} - F\para{x_0} - h f\para{x_0}}{\abs{h}},
    \]
    and we want to show that \(\epsilon \to 0\) as \(h \to 0\). We have
    \begin{align*}
        \abs{F\para{x_0 + h} - F\para{x_0} - h f\para{x_0}} &= \abs{\int_{x_0}^{x_0 + h} f\para{t} \Diff t - h \cdot f\para{x_0}} \\
        &= \abs{\int_{x_0}^{x_0 + h} \para{f\para{t} - f\para{x_0}} \Diff t} \\
        &\leq \int_{x_0}^{x_0 + h} \abs{f\para{t} - f\para{x_0}} \Diff t \\
        &\leq \sup_{t \in \ccintv{0}{h}} \abs{f\para{x_0 + t} - f\para{x_0}} \cdot \abs{h}.
    \end{align*}

    Hence,
    \[
        \abs{\epsilon\para{h}} \leq \sup_{t \in \ccintv{0}{h}} \abs{f\para{x_0 + t} - f\para{x_0}} \to 0,
    \]
    as \(t \to 0\) due to continuity, as desired.
\end{proof}

\begin{example}[Non-Differentiable Indefinite Integral]
    \label{ex:non-differentiable-indefinite-integral}%
    Take \(\ccintv{a}{b} = \ccintv{-1}{1}\), with
    \[
        f\para{x} = \begin{cases}
            -1, & x \leq 0, \\
            1, & x > 0.
        \end{cases}
    \]

    \(f\) is piecewise continuous, and \(F\) defined above is given by
    \[
        F\para{x} = \int_{-1}^{x} f\para{x} \Diff x = \begin{cases}
            -x - 1, & x \leq 0 \\
            x - 1, & x \geq 0
        \end{cases}
        = \abs{x} - 1.
    \]

    \(F\) is continuous on \(\ccintv{-1}{1}\), as expected from \autoref{prop:indef-integral-continuous}, but it is not differentiable at \(x = 0\).
\end{example}

\begin{corollary}[Integral of Continuous Derivative]
    \label{cor:integral-continuous-derivative}%
    If \(f\para{x} = g'\para{x}\) is continuous on \(\ccintv{a}{b}\), then
    \[
        F\para{x} = \int_{a}^{x} f\para{t} \Diff t = g\para{x} - g\para{a}
    \]
    for all \(x \in \ccintv{a}{b}\).
\end{corollary}

\begin{proof}
    By \autoref{thm:ftc-1} (Fundamental Theorem of Calculus), we have
    \[
        \para{F\para{x} - g\para{x}}' = 0
    \]
    on \(\ccintv{a}{b}\), and hence \(F\para{x} - g\para{x}\) is a constant, and in other words,
    \[
        F\para{x} - g\para{x} = F\para{a} - g\para{a}.
    \]

    Hence, since \(F\para{a} = 0\), we have
    \[
        F\para{x} = g\para{x} - g\para{a}
    \]
    for all \(x \in \ccintv{a}{b}\).
\end{proof}

The continuity condition is very strong -- but we can generalise the condition to a much stronger result.

\begin{theorem}[Fundamental Theorem of Calculus, Part II]
    \label{thm:ftc-2}%
    If \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) is Riemann integrable, and \(f\para{x} = F'\para{x}\) for some differentiable \(\functiondomain{F}{\ccintv{a}{b}}{\RR}\), then we have
    \[
        \int_{a}^{b} f\para{t} \Diff t = \int_{a}^{b} F'\para{t} \Diff t = F\para{b} - F\para{a}.
    \]
\end{theorem}

\begin{proof}
    Let \(\partition\) be any partition. Applying \autoref{thm:mean-value-thm} (Mean Value Theorem) to \(F\) on intervals of this partition gives
    \[
        F\para{x_j} - F\para{x_{j - 1}} = f\para{t_j} \cdot \para{x_j - x_{j - 1}}
    \]
    for some \(t_j \in \oointv{x_j}{x_{j - 1}}\).

    Summing over all \(j\)s, we have
    \[
        F\para{b} - F\para{a} = \sum_{j = 1}^{n} f\para{t_j} \para{x_j - x_{j - 1}}.
    \]

    Since we have \(\inf_{x \in \ccintv{x_{j - 1}}{x_j}} f\para{x} \leq f\para{t_j} \leq \sup_{x \in \ccintv{x_{j - 1}}{x_j}} f\para{x}\), this means
    \[
        \lowerSum\para{f, \partition} \leq F\para{b} - F\para{a} \leq \upperSum\para{f, \partition}.
    \]

    Hence, we must have
    \[
        \lowerI\para{f} \leq F\para{b} - F\para{a} \leq \upperI\para{f},
    \]
    and since \(f\) is integrable, we have
    \[
        I\para{f} = F\para{b} - F\para{a},
    \]
    as desired.
\end{proof}

Now, we may use this, with our known differentiation rules, to derive some (already familiar) integration rules.

\begin{proposition}[Integration by Parts]
    \label{prop:integration-by-parts}%
    Let \(f, g \in \contClass{1} \para{\ccintv{a}{b}}\) be continuously differentiable. Then
    \[
        \int_{a}^{b} f'\para{x} g\para{x} \Diff x = f\para{b} g\para{b} - f\para{a} g\para{a} - \int_{a}^{b} f\para{x} g'\para{x} \Diff x.
    \]
\end{proposition}

\begin{proof}
    This follows from \autoref{thm:ftc-2} (Fundamental Theorem of Calculus) and by product rule. We have
    \[
        f'\para{x} g\para{x} = \para{f\para{x} g\para{x}}' - f\para{x} g'\para{x}.
    \]

    Integrate over \(\ccintv{a}{b}\), and using \autoref{thm:ftc-2} gives the result as desired.
\end{proof}

\begin{proposition}[Integration by Substitution]
    \label{prop:integration-by-substitution}%
    Let \(f \in \contClass{0} \para{\ccintv{a}{b}}\) be continuous, and \(\functiondomain{g}{\ccintv{\alpha}{\beta}}{\ccintv{a}{b}}\) be continuously differentiable, with \(g\para{\alpha} = a\), and \(g\para{\beta} = b\). Then
    \[
        \int_{a}^{b} f\para{x} \Diff x = \int_{\alpha}^{\beta} f\para{g\para{t}} g'\para{t} \Diff t.
    \]
\end{proposition}

\begin{proof}
    Set
    \[
        F\para{x} = \int_{a}^{x} f\para{t} \Diff t.
    \]
    
    Then \(F\) is well-defined and differentiable, by \autoref{thm:ftc-1} (Fundamental Theorem of Calculus). Set \(\functiondomain{h = F \circ g}{\ccintv{\alpha}{\beta}}{\RR}\). Then \(h\) is differentiable, and by \autoref{prop:chain-rule} (Chain Rule), we have
    \[
        h'\para{t} = F'\para{g\para{t}} g'\para{t}.
    \]

    Hence, by applying \autoref{thm:ftc-2} twice, we have
    \begin{align*}
        \int_{a}^{b} f\para{x} \Diff x &= F\para{b} - F\para{a} \\
        &= F\para{g\para{\beta}} - F\para{g\para{\alpha}} \\
        &= h\para{\beta} - h\para{\alpha} \\
        &= \int_{\alpha}^{\beta} h'\para{t} \Diff t \\
        &= \int_{\alpha}^{\beta} f\para{g\para{t}} g'\para{t} \Diff t.
    \end{align*}
\end{proof}

A very important application of this is providing another remainder for Taylor's Theorem.

\begin{theorem}[Taylor's Theorem with Integral Remainder]
    \label{thm:taylor-thm-integral-remainder}%
    Let \(\functiondomain{f}{\ccintv{a}{a + h}}{\RR}\) be \(n\) times continuously differentiable on \(\oointv{a}{a + h}\). Then the \emph{Integral Form of the Remainder} gives the Taylor remainder in the form
    \begin{align*}
        R_{n, f, a} \para{h} &= \frac{h^{n}}{\para{n - 1}!} \cdot \int_{0}^{1} \para{1 - t}^{n - 1} f^{\para{n}} \para{a + th} \Diff t \\
        &= \frac{1}{\para{n - 1}!} \cdot \int_{0}^{h} \para{h - u}^{n - 1} f^{\para{n}} \para{a + u} \Diff u.
    \end{align*}
\end{theorem}

\begin{remark}
    By \autoref{thm:extreme-value} (Extreme Value Theorem), there is some \(M_n = \sup_{x \in \ccintv{0}{h}} \abs{f^{\para{n}} \para{a + h}}\), since \(f^{\para{n}}\) is continuous. Then, we have
    \[
        \abs{R_{n, f, a} \para{h}} \leq \abs{\frac{h^n}{\para{n - 1}!} \cdot M_n \cdot \int_{0}^{1} \para{1 - t}^{n - 1} \Diff t} = \frac{\abs{h^n}}{n!} \cdot M_n
    \]

    If we knew \(\sup_{n \geq 0} M_n = M < \plusinfty\), we have
    \[
        \abs{R_{n, f, a} \para{h}} \leq \frac{M \abs{h^n}}{n!} \to 0
    \]
    as \(n \to \infty\), for all \(\abs{h} < 1\). Therefore, \(f\) is analytic around \(a\).
\end{remark}

\begin{remark}[on Complex Differentiability]
    This version generalises to complex functions as well. However, the reason we do not discuss much differentiability and Taylor Theorems in \(
    \CC\), is because if the function is differentiable over \(\CC\), then it is smooth, and this comes with the estimates on \(M_n\), telling us that all smooth functions on \(\CC\) are analytic.
\end{remark}

\begin{proof}
    Using \autoref{prop:integration-by-parts} (Integration by Parts), we have
    \begin{align*}
        &\phantom{=} \frac{1}{\para{n - 1}!} \cdot \int_{0}^{h} \para{h - u}^{n - 1} f^{\para{n}} \para{a + u} \Diff u \\
        &= - h^{n - 1} \cdot \frac{f^{\para{n - 1}} \para{a}}{\para{n - 1}!} + \frac{1}{\para{n - 2}!} \int_{0}^{h} \para{h - u}^{n - 2} f^{\para{n - 1}} \para{a + u} \Diff u \\
        &= \cdots \\
        &= - \sum_{k = 1}^{n - 1} \frac{f^{\para{k}} \para{a}}{k!} \cdot h^k + \int_{0}^{h} f'\para{a + u} \Diff u \\
        &= - \sum_{k = 0}^{n - 1} \frac{f^{\para{k}} \para{a}}{k!} \cdot h^k + f\para{a + h},
    \end{align*}
    as desired.
\end{proof}

We may start from this point and give alternative proofs of the mean-value form of the remainders.

\begin{theorem}[Mean Value Theorem for Integrals]
    \label{thm:mvt-integrals}%
    Let \(\functiondomain{f}{\ccintv{a}{b}}{\RR}\) be continuous. Then there is some \(c \in \oointv{a}{b}\), such that
    \[
        \int_{a}^{b} f\para{x} \Diff x = f\para{c} \cdot \para{b - a}.
    \]
\end{theorem}

\begin{proof}
    Applying \autoref{thm:mean-value-thm} (Mean Value Theorem) to the function
    \[
        F\para{x} = \int_{a}^{x} f\para{t} \Diff t,
    \]
    there is some \(c \in \oointv{a}{b}\), giving
    \[
        \frac{F\para{b} - F\para{a}}{b - a} = F'\para{c}.
    \]

    Using \autoref{thm:ftc-1} (Fundamental Theorem of Calculus), we may get the result as desired.
\end{proof}

There is a more general version of this theorem, just like how Cauchy's Mean Value Theorem is more general than the Mean Value Theorem.

\begin{theorem}[Cauchy Mean Value Theorem for Integrals]
    \label{thm:cauchy-mvt-integrals}%
    Let \(\functiondomain{f, g}{\ccintv{a}{b}}{\RR}\) be continuous, and \(g\para{x} \neq 0\) for all \(x \in \ccintv{a}{b}\). Then there is some \(c \in \oointv{a}{b}\), such that
    \[
        \int_{a}^{b} f\para{x} g\para{x} \Diff x = f\para{c} \cdot \int_{a}^{b} g\para{x} \Diff x.
    \]
\end{theorem}

\begin{proof}
    Applying \autoref{thm:cauchy-mvt} (Cauchy Mean Value Theorem) to the functions
    \[
        F\para{x} = \int_{a}^{x} f\para{t} g\para{t} \Diff t \quad \text{and} \quad G\para{x} = \int_{a}^{x} g\para{t} \Diff t,
    \]
    there is some \(c \in \oointv{a}{b}\), giving
    \[
        \para{F\para{b} - F\para{a}} G'\para{c} = F'\para{c} \para{G\para{b} - G\para{a}}.
    \]

    Hence, using \autoref{thm:ftc-1} (Fundamental Theorem of Calculus), we may simplify this result, as desired.
\end{proof}

We prove the two forms of the Lagrange remainder, but with slightly stronger conditions.

\begin{proof}[of \autoref{thm:taylor-thm-langrange-remainder} (Lagrange Form of the Remainder), assuming continuity of \(f^{\para{n}}\)]
    Let \(g\para{t} = \para{1 - t}^{n - 1}\). Then using \autoref{thm:taylor-thm-integral-remainder} (Integral Form of the Remainder) and \autoref{thm:cauchy-mvt-integrals} (Cauchy Mean Value Theorem for Integrals), there is some \(\theta \in \oointv{0}{1}\), such that
    \begin{align*}
        R_{n, f, a} \para{h} &= \frac{h^n}{\para{h - 1}!} f^{\para{n}} \para{a + \theta h} \int_{0}^{1} \para{1 - t}^n \Diff t \\
        &= \frac{h^n}{n!} \cdot f^{\para{n}} \para{a + \theta h}.
    \end{align*}
\end{proof}

\begin{proof}[of \autoref{thm:taylor-thm-cauchy-remainder} (Cauchy Form of the Remainder), assuming continuity of \(f^{\para{n}}\)]
    Let \(g\para{t} = 1\) be constant. Then using \autoref{thm:taylor-thm-integral-remainder} (Integral Form of the Remainder) and \autoref{thm:mvt-integrals} (Mean Value Theorem for Integrals), there is some \(\theta \in \oointv{0}{1}\), such that
    \begin{align*}
        R_{n, f, a} \para{h} &= \frac{h^n}{\para{h - 1}!} \para{1 - \theta}^{h} f^{\para{n}} \para{a + \theta h} \int_{0}^{1} \Diff t \\
        &= \frac{h^n}{\para{n - 1}!} \cdot \para{1 - \theta}^{n - 1} \cdot f^{\para{n}} \para{a + \theta h}.
    \end{align*}
\end{proof}

\begin{proof}[of \autoref{thm:taylor-thm-schlomilch-remainder} (Schl\"{o}milch Form of the Remainder), asusming continuitiy of \(f^{\para{n}}\)]
    Let \(g\para{t} = \para{1 - t}^{p - 1}\). Then using \autoref{thm:taylor-thm-integral-remainder} (Integral Form of the Remainder), and \autoref{thm:cauchy-mvt-integrals} (Cauchy Mean Value Theorem for Integrals), there is some \(\theta \in \oointv{0}{1}\), such that
    \begin{align*}
        R_{n, f, a} \para{h} &= \frac{h^n}{\para{n - 1}!} \cdot \int_{0}^{1} \para{1 - t}^{n - 1} f^{\para{n}} \para{a + th} \Diff t \\
        &= \frac{h^n}{\para{n - 1}!} \cdot \int_{0}^{1} \brac{f^{\para{n}} \para{a + th} \cdot \para{1 - t}^{n - p}} \cdot \para{1 - t}^{p - 1} \Diff t \\
        &= \frac{h^n}{\para{n - 1}!} \cdot f^{\para{n}} \para{a + \theta h} \cdot \para{1 - \theta}^{n - p} \cdot \int_{0}^{1} \para{1 - t}^{p - 1} \Diff t \\
        &= \frac{h^n}{p \para{n - 1}!} \cdot \para{1 - \theta}^{n - p} f^{\para{n}} \para{a + \theta h}.
    \end{align*}
\end{proof}

\subsection{Improper Integrals}

\subsubsection{Integrals over Unbounded Domain}

\begin{definition}[Improper Integrals over an Unbounded Domain]
    Suppose \(\functiondomain{f}{\cointv{a}{\plusinfty}}{\RR}\) is integrable on \(\ccintv{a}{R}\), for all \(R \in \cointv{a}{\plusinfty}\). We let
    \[
        \function{F}{\cointv{a}{\plusinfty}}{\RR}{R}{\int_{a}^{R} f\para{x} \Diff x}.
    \]

    Then we say the \emph{improper integral}
    \[
        \int_{a}^{\plusinfty} f\para{x} \Diff x
    \]
    exists/converges, if \(L = \lim_{R \to \plusinfty} F\para{x}\) exists and is finite, and we set
    \[
        \int_{a}^{\plusinfty} f\para{x} \Diff x = L.
    \]

    Similarly, we may define \(\int_{\minusinfty}^{b} g\para{x} \Diff x\).

    If \(\functiondomain{f}{\RR}{\RR}\) is such that \(\int_{a}^{\plusinfty} f\para{x} \Diff x = L_1\) and \(\int_{\minusinfty}^{a} f\para{x} \Diff x = L_2\), for some \(a \in \RR\), then we say the improper integral
    \[
        \int_{\minusinfty}^{\plusinfty} f\para{x} \Diff x
    \]
    exists, and we set
    \[
        \int_{\minusinfty}^{\plusinfty} f\para{x} \Diff x = \int_{\minusinfty}^{a} f\para{x} \Diff x + \int_{a}^{\plusinfty} f\para{x} \Diff x = \lim_{R_1 \to \plusinfty} \int_{a}^{R_1} f\para{x} \Diff x + \lim_{R_2 \to \minusinfty} \int_{R_2}^{a} f\para{x} \Diff x.
    \]
\end{definition}

\begin{remark}
    This is different from taking the limit
    \[
        \lim_{R \to \plusinfty} \brac{\int_{-R}^{R} f\para{x} \Diff x}.
    \]

    If the above improper integral exists, then this new limit (the Cauchy Principle Value) exists and is equal -- but not necessarily otherwise.
\end{remark}

\begin{examples}[Improper Integrals over Unbounded Intervals]
    \label{ex:improper-integrals-unbounded-intervals}%
    \begin{itemize}
        \item The integral
        \[
            \int_{1}^{\plusinfty} \frac{1}{x^p} \Diff x = \lim_{R \to \plusinfty} \int_{1}^{R} \frac{1}{x^p} \Diff x
        \]
        exists, if and only if \(p > 1\).

        \item We have
        \[
            \int_{2}^{\plusinfty} \frac{1}{x \log^2 x} \Diff x = \lim_{R \to \plusinfty} \int_{2}^{R} \frac{\Diff x}{x \log^2 x} = \frac{1}{\log 2}.
        \]

        \item The \emph{Gaussian Integral} is given by
        \[
            \int_{\minusinfty}^{\plusinfty} e^{- x^2} \Diff x.
        \]

        However, its antiderivative is difficult to calculate (and is in fact non-elementary). We can use alternative methods to show this converges.
    \end{itemize}
\end{examples}

\subsubsection{Tests on Convergence of Improper Integrals}

We introduce some tests for improper integrals. However, the most obvious test for infinite series in fact fails for improper integrals!

\begin{remark}
    Recall the \(n\)th term test for series, introduced in \autoref{prop:ser-test-nth-term}. However, this does not necessarily work for improper integrals, even if the function is continuous. Formally,
    \[
        \int_{a}^{\plusinfty} f\para{x} \Diff x \text{ exists} \notimplies \lim_{x \to \plusinfty} f\para{x} = 0.
    \]

    Indeed, if the limit exists, then it must be zero, but the limit does necessarily exist.
\end{remark}

We first investigate the convergence of improper integrals of non-negative functions, so we introduce \autoref{prop:fn-monotone-convergence-theorem} for monotone functions.

\begin{proposition}[Monotone Convergence Theorem for Functions]
    \label{prop:fn-monotone-convergence-theorem}%
    Let \(\functiondomain{f}{\cointv{a}{\plusinfty}}{\RR}\) be increasing.

    \begin{itemize}
        \item If \(f\) is bounded above, then we have
        \[
            \lim_{x \to \plusinfty} f\para{x} \Diff x = \sup_{x \geq a} f\para{x}.
        \]
        \item If \(f\) is not bounded above, then we have
        \[
            \lim_{x \to \plusinfty} f\para{x} \Diff x = \plusinfty.
        \]
    \end{itemize}

    Similarly, let \(\functiondomain{g}{\cointv{a}{\plusinfty}}{\RR}\) be decreasing.

    \begin{itemize}
        \item If \(g\) is bounded below, then we have
        \[
            \lim_{x \to \plusinfty} g\para{x} \Diff x = \inf_{x \geq a} g\para{x}.
        \]
        \item If \(g\) is not bounded below, then we have
        \[
            \lim_{x \to \plusinfty} g\para{x} \Diff x = \minusinfty.
        \]
    \end{itemize}
\end{proposition}

\begin{proof}
    We prove this for increasing functions, and the decreasing case is similar.

    Let \(\epsilon > 0\). If \(f\para{x}\) is bounded above, by the definition of the supremum, there is some \(R \in \cointv{a}{\plusinfty}\), such that
    \[
        \sup_{x \geq a} f\para{x} - \epsilon < f\para{R} \leq \sup_{x \geq a} f\para{x}.
    \]

    This means that for all \(x > R\), we have
    \[
        \sup_{x \geq a} f\para{x} - \epsilon < f\para{R} \leq f\para{x} \leq \sup_{x \geq a} f\para{x},
    \]
    so in particular,
    \[
        \abs{f\para{x} - \sup_{x \geq a} f\para{x}} < \epsilon,
    \]
    showing the limit as desired.

    However, if \(f\para{x}\) is not bounded above, for any threshold \(M > 0\), there is some \(R \in \cointv{a}{\plusinfty}\), such that
    \[
        f\para{R} > M.
    \]

    Hence, for all \(x > R\), we have
    \[
        f\para{x} \geq f\para{R} > M,
    \]
    which shows divergence to \(\plusinfty\) as desired.
\end{proof}

Similar to the Cauchy characterisation (\autoref{thm:completeness-of-r-and-c}) of the limit of a sequence, we may characterise the limit of a function similarly as well.

\begin{proposition}[Cauchy Criteria for Function Limit at Infinity]
    \label{prop:cauchy-criteria-funciton-limit-infinity}%
    Let \(\functiondomain{f}{X \subseteq \CC}{\CC}\) be such that \(\infty\) is an accumulation point of \(X\). Then \(f\para{x}\) converges to some finite limit as \(x \to \infty\), if and only if
    \[
        \forall \epsilon > 0, \exists R > 0, \forall x_1, x_2 \in X, \abs{x_1}, \abs{x_2} > R: \abs{f\para{x_1} - f\para{x_2}} < \epsilon.
    \]
\end{proposition}

\begin{proof}
    Suppose that \(f\para{x} \to L \in \CC\) as \(x \to \infty\), and let \(\epsilon > 0\) be arbitrary. By the definition of a limit, for some \(R > 0\), we have
    \[
        \forall x \in X, \abs{x} > R: \abs{f\para{x} - L} < \frac{\epsilon}{2}.
    \]

    Hence, for all \(x_1, x_2 \in X, \abs{x_1}, \abs{x_2} > R\), we have
    \begin{align*}
        \abs{f\para{x_1} - f\para{x_2}} &\leq \abs{f\para{x_1} - L} + \abs{f\para{x_2} - L} \\
        &< \frac{\epsilon}{2} + \frac{\epsilon}{2} \\
        &= \epsilon
    \end{align*}
    as desired.

    On the other hand, suppose that the Cauchy criteria holds. Let \(\para{z_n}_{n \in \NN}\) be a sequence on \(X\) with limit \(\infty\). Then, the sequence \(\para{f\para{z_n}}_{n \in \NN}\) is Cauchy, and hence tends to some limit \(L \in \CC\).

    Now, let \(\epsilon > 0\) be arbitrary, and let \(R > 0\) be such that
    \[
        \forall x_1, x_2 \in X, \abs{x_1}, \abs{x_2} > R: \abs{f\para{x_1} - f\para{x_2}} < \frac{\epsilon}{2},
    \]
    and let \(n > 0\) be sufficiently large, such that
    \[
        \abs{z_n} > R \quad \text{and} \quad \abs{f\para{z_n} - L} < \frac{\epsilon}{2}
    \]  

    For any \(x \in X, \abs{x} > R\), we have
    \begin{align*}
        \abs{f\para{x} - L} &\leq \abs{f\para{x} - f\para{z_n}} + \abs{f\para{z_n} - L} \\
        &< \frac{\epsilon}{2} + \frac{\epsilon}{2} \\
        &= \epsilon
    \end{align*}
    as desired.
\end{proof}

Now we introduce the tests.

\begin{proposition}[Comparison Test for Integrals]
    \label{prop:int-test-comparison}%
    Let \(\functiondomain{f, g}{\cointv{a}{\plusinfty}}{\RR}\) be integrable over any finite interval \(\ccintv{a}{R}\) for \(R \geq a\), and satisfies that \(0 \leq f\para{x} \leq g\para{x}\) for all \(x \in \cointv{a}{\plusinfty}\). Then the following holds.

    \begin{enumerate}
        \item If the improper integral of \(g\) to infinity exists, then so does the improper integral of \(f\), and we have
        \[
            \int_{a}^{\plusinfty} f\para{x} \Diff x \leq \int_{a}^{\plusinfty} g\para{x} \Diff x.
        \]

        \item If the improper integral of \(f\) to infinity diverges (to infinity), then so does the improper integral of \(g\).
    \end{enumerate}
\end{proposition}

\begin{proof}
    Let
    \[
        F\para{R} = \int_{a}^{R} f\para{x} \Diff x \quad \text{and} \quad G\para{R} = \int_{a}^{R} g\para{x} \Diff x.
    \]

    Note that \(F\) and \(G\) are both increasing, with
    \[
        0 \leq F\para{R} \leq G\para{R}.
    \]

    \begin{enumerate}
        \item Since \(G\) converges, it is bounded above, and therefore so is \(F\). Hence, \(F\) converges, and since limit preserve weak inequalities, we have
        \[
            \lim_{R \to \plusinfty} F\para{R} = \int_{a}^{\plusinfty} f\para{x} \Diff x \leq \int_{a}^{\plusinfty} g\para{x} \Diff x = \lim_{R \to \plusinfty} G\para{R}.
        \]
        

        \item Since \(f \geq 0\), from \autoref{prop:fn-monotone-convergence-theorem}, we have
        \[
            \lim_{R \to \plusinfty} F\para{R} = \plusinfty
        \]
        necessarily, and in particular, \(F\) is not bounded above.

        But we have \(G\para{R} \geq F\para{R}\), so \(G\) is not bounded above either, and hence
        \[
            \lim_{R \to \plusinfty} G\para{R} = \int_{a}^{\plusinfty} g\para{x} \Diff x = \plusinfty.
        \]
    \end{enumerate}
\end{proof}

\begin{example}[Application of Direct Comparison Test]
    \label{ex:app-direct-comparison-test}%
    We return to the Gaussian integral. For \(x \geq 1\), we have \(x^2 \geq x\), so \(e^{-x^2} \leq e^{-x}\).

    Therefore, we have
    \[
        \int_{1}^{\plusinfty} e^{-x^2} \Diff x \leq \int_{1}^{\plusinfty} e^{-x} \Diff x = \brac{- \frac{1}{e^x}}^{\plusinfty}_{1} = \frac{1}{e}.
    \]

    Hence,
    \[
        \int_{\minusinfty}^{\plusinfty} e^{-x^2} \Diff x = 2 \int_{0}^{\plusinfty} e^{-x^2} \Diff x = 2 \int_{1}^{\plusinfty} e^{-x^2} \Diff x + 2 \int_{0}^{1} e^{-x^2} \Diff x
    \]
    so the improper integral converges, as desired.
\end{example}

There is a similar notion of Cauchy sequences in improper integrals.

\begin{proposition}[Cauchy-Ness is Equivalent to Convergence of Improper Integrals]
    \label{prop:int-cauchy}%
    Let \(\functiondomain{f}{\cointv{a}{\plusinfty}}{\RR}\) be integrable on \(\ccintv{a}{R}\) for any \(R \geq a\). Then the improper integral
    \[
        \int_{a}^{\plusinfty} f\para{x} \Diff x
    \]
    converges, if and only if for any \(\epsilon > 0\), there is some \(R \geq a\), such that for all \(R \leq x_1 \leq x_2\), we have
    \[
        \abs{\int_{x_1}^{x_2} f\para{x} \Diff x} < \epsilon.
    \]
\end{proposition}

\begin{proof}
    This is a direct consequence of \autoref{prop:cauchy-criteria-funciton-limit-infinity} (Cauchy Criteria for Function Limit at Infinity).
\end{proof}

Similar to when we did series, there is a notion of `absolute convergence' for integrals -- which allows us to extend the above tests to improper integrals of functions whose values are not necessarily positive.

\begin{proposition}[Absolute Convergence implies Convergence for Improper Integrals]
    \label{prop:int-absolute-convergence}%
    Let \(\functiondomain{f}{\cointv{a}{\plusinfty}}{\RR}\) be integrable on \(\ccintv{a}{R}\) for any \(R \geq a\). Then if the improper integral of its absolute value
    \[
        \int_{a}^{\plusinfty} \abs{f\para{x}} \Diff x
    \]
    converges, then so does the improper integral
    \[
        \int_{a}^{\plusinfty} f\para{x} \Diff x.
    \]
\end{proposition}

\begin{proof}[using \autoref{prop:int-test-comparison} (Comparison Test)]
    We note that we have
    \[
        0 \leq f\para{x} + \abs{f\para{x}} \leq 2 \abs{f\para{x}},
    \]
    so the result follow from \autoref{prop:int-test-comparison} (Comparison Test for Improper Integrals).
\end{proof}

\begin{proof}[using \autoref{prop:int-cauchy} (Cauchy is equivalent to Convergence)]
    Let
    \[
        F\para{x} = \int_{a}^{x} f\para{t} \Diff t, \quad G\para{x} = \int_{a}^{x} \abs{f\para{t}} \Diff t.
    \]

    Hence, we have for \(x_1, x_2 \geq a\).
    \begin{align*}
        \abs{F\para{x_2} - F\para{x_1}} &= \abs{\int_{x_1}^{x_2} f\para{t} \Diff t} \\
        &\leq \int_{x_1}^{x_2} \abs{f\para{t}} \Diff t \\
        &= G\para{x_2} - G\para{x_1}.
    \end{align*}

    This shows that if \(G\) Cauchy, then \(F\) is as well. But by \autoref{prop:int-cauchy}, this means if \(G\) converges, then \(F\) converges as well, finishing the proof.
\end{proof}

The final test is the Dirichlet test, which investigates the integral of the product of two functions.

\begin{proposition}[Dirichlet Test for Improper Integrals]
    \label{prop:int-test-dirichlet}%
    Let \(f \in \contClass{0} \para{\cointv{a}{\plusinfty}}\) be continuous, and suppose it has a bounded antiderivative
    \[
        F\para{x} = \int_{a}^{x} f\para{t} \Diff t.
    \]

    Let \(g \in \contClass{1} \para{\cointv{a}{\plusinfty}}\) be continuously differentiable, monotone, and satisfies
    \[
        \lim_{x \to \plusinfty} g\para{x} = 0.
    \]

    Then the improper integral
    \[
        \int_{a}^{\plusinfty} f\para{x} g\para{x} \Diff x
    \]
    converges.
\end{proposition}

Similar to the proofs of \autoref{prop:ser-test-dirichlet} (Dirichlet Test for Infinite Series) using \autoref{lem:ser-summation-by-parts} (Summation by Parts), we use \autoref{prop:integration-by-parts} (Integration by Parts).

\begin{proof}[using \autoref{prop:int-absolute-convergence} (Absolute Convergence)]
    We let
    \[
        F\para{x} = \int_{a}^{x} f\para{t} \Diff t, \quad H\para{x} = \int_{a}^{x} f\para{t} g\para{t} \Diff t.
    \]

    Both \(F\) and \(G\) are continuously differentiable. Hence, using  integration by parts, we have
    \begin{align*}
        H\para{x} &= \int_{a}^{x} F'\para{t} g\para{t} \Diff t \\
        &= F\para{x} g\para{x} - F\para{a} g\para{a} - \int_{a}^{x} F\para{t} g'\para{t} \Diff t
    \end{align*}

    Let \(M = \sup_{x \geq a} \abs{F\para{x}}\). For the final term, we have the integral of the absolute value is
    \begin{align*}
        \int_{a}^{x} \abs{F\para{t} g'\para{t}} \Diff t &\leq \sup_{x \geq a} \abs{F\para{t}} \cdot \int_{a}^{x} \abs{g'\para{t}} \Diff t \\
        &= \pm M \cdot \int_{a}^{x} g'\para{t} \Diff t \\
        &= M \cdot \abs{g\para{a} - g\para{x}}.
    \end{align*}

    Hence, as \(x \to \plusinfty\), we have
    \[
        \int_{a}^{\plusinfty} \abs{F\para{t} g'\para{t}} \Diff t \leq M \cdot \abs{g\para{a}},
    \]
    so the improper integral of  \(\abs{F\para{t} g'\para{t}}\) converges, and hence the improper integral of \(F\para{t} g'\para{t}\) converges, by \autoref{prop:int-absolute-convergence} (Absolute Convergence of Improper Integral).

    Hence, as \(x \to \infty\), \(F\para{x}\) is bounded while \(g\para{x} \to 0\), so \(F\para{x} g\para{x} \to 0\), and the final improper integral converges. So \(H\) converges, as desired.
\end{proof}

\subsubsection{Integrals over Isolated Singularity}

We now investigate another type of improper integrals.

\begin{definition}[Improper Integrals over Isolated Singularity]
    \label{def:improper-integral-isolated-singularity}%
    Let \(\functiondomain{f}{\ocintv{a}{b}}{\RR}\), and suppose that \(f\) is integrable over all \(\ccintv{a + \delta}{b}\) for all \(0 < \delta \leq b - a\). Then, let
    \[
        \function{F}{\ocintv{0}{b - a}}{\RR}{\delta}{\int_{a + \delta}^{b} f\para{x} \Diff x}.
    \]

    We say the \emph{improper integral} \(\int_{a}^{b} f\para{x} \Diff x\) exists/converges, if the limit
    \[
        \lim_{\delta \to 0^+} F\para{\delta}
    \]
    exists, and we denote this value as
    \[
        \int_{a}^{b} f\para{x} \Diff x = \lim_{\delta \to 0^+} F\para{\delta},
    \]
    and we say it does not exist/diverges otherwise.

    We may similarly define this for \(\functiondomain{f}{\cointv{a}{b}}{\RR}\).

    Let \(\functiondomain{f}{\ccintv{B}{b} \setminus \brce{a}}{\RR}\) is such that
    \[
        \int_{B}^{a} f\para{x} \Diff x = l_1 \quad \text{and} \quad \int_{a}^{b} f\para{x} \Diff x = l_2
    \]
    both converges, then we say the improper integral \(\int_{B}^{b} f\para{x} \Diff x\) exists, and we denote this value as
    \[
        \int_{B}^{b} f\para{x} \Diff x = l_1 + l_2.
    \]
\end{definition}

\begin{remark}
    Similar to the improper integral from \(\minusinfty\) to \(\plusinfty\) case, we are in fact taking two separate limits
    \[
        \int_{B}^{b} f\para{x} \Diff x = \lim_{\delta_1 \to 0^+} \int_{B}^{a - \delta} f\para{x} \Diff x + \lim_{\delta_2 \to 0^+} \int_{a + \delta}^{b} f\para{x} \Diff x.
    \]

    (The limit where we take both limits simultaneously is called the Cauchy Principle Value.)
\end{remark}

\begin{examples}[Improper Integrals over Isolated Singularity]
    \label{ex:improper-integral-isolated-singularity}%
    \begin{itemize}
        \item We investigate the convergence of the improper integral \(\int_{0}^{1} \frac{1}{x^p} \Diff x\) for \(p > 0\). Note that
        \[
            \int_{\delta}^{1} \frac{1}{x^p} \Diff x = \begin{cases}
                \log 1 - \log \delta, & p = 1, \\
                \frac{\delta^{1 - p} - 1}{1 - p}, & p \neq 1.
            \end{cases}
        \]

        So this integral converges if and only if \(p < 1\).

        \item We have, using the substitution \(u = \log\para{1 / x}\), that
        \[
            \int_{0}^{\frac{1}{2}} \frac{\Diff x}{x \log\para{1 / x}^p} = \int_{\plusinfty}^{\log 2} \para{- \frac{1}{u^p}} \Diff u = \int_{\log 2}^{\plusinfty} \frac{1}{u^p} \Diff u
        \]
        converges if and only if \(p > 1\).
    \end{itemize}
\end{examples}

From the examples, we see that we may use some substitution to change the improper integral with the isolated point to some improper integral to infinity. This is formally expressed in \autoref{lem:conversion-types-improper-integrals}, but this should only be regarded as a broad idea rather than a concrete lemma.

\begin{lemma}[Conversion between Types of Improper Integrals]
    \label{lem:conversion-types-improper-integrals}%
    Let \(\functiondomain{f}{\ocintv{a}{b}}{\RR}\) be Riemann integrable on \(\cointv{a + \delta}{b}\) for all \(0 < \delta < b - a\). For some continuously differentiable bijection
    \[
        \functiondomain{\phi}{\cointv{c}{\plusinfty}}{\ocintv{a}{b}}
    \]
    which is strictly decreasing with \(\phi\para{c} = b\) and \(\lim_{t \to \plusinfty} \phi\para{t} = a\). Then
    \[
        \int_{a}^{b} f\para{x} \Diff x = \int_{c}^{\plusinfty} f\para{\phi\para{t}} \para{- \phi'\para{t}} \Diff t,
    \]
    in the sense that they either both converge (to the same value) or both diverge.
\end{lemma}
